    <!-- Script da Aplicação -->
    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // Elementos da UI
            const views = {
                selector: document.getElementById('selector-view'),
                questionnaire: document.getElementById('questionnaire-view'),
                results: document.getElementById('results-view'),
            };
            const searchInput = document.getElementById('search-input');
            const filterSelect = document.getElementById('filter-select');
            const questionnaireList = document.getElementById('questionnaire-list');
            const backBtn = document.getElementById('back-to-selector-btn');
            const calculateBtn = document.getElementById('calculate-score-btn');
            const newQuestionnaireBtn = document.getElementById('new-questionnaire-btn');
            const printBtn = document.getElementById('print-results-btn');
            const formError = document.getElementById('form-error');

            // Carrega os dados JSON
            const promsDataRaw = JSON.parse(document.getElementById('proms-data').textContent);
            const allQuestionnaires = Object.values(promsDataRaw).flat();

            // Estado da aplicação
            let currentQuestionnaire = null;
            let patientAnswers = {};

            // Função para mudar a visualização
            const switchView = (viewName) => {
                Object.values(views).forEach(view => view.classList.add('hidden'));
                if (views[viewName]) {
                    views[viewName].classList.remove('hidden');
                    window.scrollTo(0, 0);
                }
            };
            
            // Renderiza a lista de questionários
            const renderQuestionnaireList = (questionnaires) => {
                questionnaireList.innerHTML = '';
                if (questionnaires.length === 0) {
                    questionnaireList.innerHTML = `<p class="text-gray-500 col-span-full text-center">Nenhum questionário encontrado.</p>`;
                    return;
                }
                questionnaires.forEach(q => {
                    const card = document.createElement('div');
                    card.className = 'bg-white p-5 rounded-lg shadow-md hover:shadow-xl hover:-translate-y-1 transition-all cursor-pointer';
                    card.innerHTML = `
                        <h3 class="font-bold text-lg text-teal-700">${q.sigla}</h3>
                        <p class="text-gray-600 text-sm">${q.nome_completo_pt}</p>
                        <p class="text-xs text-gray-400 mt-2">${q.descricao}</p>
                    `;
                    card.addEventListener('click', () => startQuestionnaire(q.sigla));
                    questionnaireList.appendChild(card);
                });
            };

            // Filtra e busca os questionários
            const filterAndSearch = () => {
                const searchTerm = searchInput.value.toLowerCase().trim();
                const filterValue = filterSelect.value;
                
                const filtered = allQuestionnaires.filter(q => {
                    const matchesFilter = filterValue === 'all' || promsDataRaw[filterValue]?.some(item => item.sigla === q.sigla);
                    
                    const matchesSearch = searchTerm === '' ||
                        q.sigla.toLowerCase().includes(searchTerm) ||
                        q.nome_completo_pt.toLowerCase().includes(searchTerm) ||
                        q.descricao.toLowerCase().includes(searchTerm) ||
                        (q.keywords_pt && q.keywords_pt.some(k => k.toLowerCase().includes(searchTerm)));

                    return matchesFilter && matchesSearch;
                });
                renderQuestionnaireList(filtered);
            };

            // Inicia um questionário
            const startQuestionnaire = (sigla) => {
                currentQuestionnaire = allQuestionnaires.find(q => q.sigla === sigla);
                if (!currentQuestionnaire) return;

                patientAnswers = {};
                document.getElementById('questionnaire-title').textContent = currentQuestionnaire.nome_completo_pt;
                document.getElementById('questionnaire-instructions').textContent = currentQuestionnaire.conteudo.instrucoes;
                
                const form = document.getElementById('questionnaire-form');
                form.innerHTML = '';
                
                currentQuestionnaire.conteudo.perguntas.forEach((p, index) => {
                    const questionBlock = document.createElement('div');
                    questionBlock.className = 'mb-6 p-4 rounded-md bg-gray-50';
                    let optionsHTML = `<div class="flex flex-wrap gap-x-6 gap-y-2 mt-2">`;
                    currentQuestionnaire.conteudo.opcoes_resposta.forEach(opt => {
                        optionsHTML += `
                            <label class="inline-flex items-center cursor-pointer">
                                <input type="radio" name="q${p.id_pergunta}" value="${opt.valor}" class="form-radio text-teal-600" required>
                                <span class="ml-2 text-gray-700">${opt.texto}</span>
                            </label>
                        `;
                    });
                    optionsHTML += `</div>`;
                    
                    questionBlock.innerHTML = `
                        <p class="font-semibold text-gray-800">${index + 1}. ${p.texto_pergunta}</p>
                        ${optionsHTML}
                    `;
                    form.appendChild(questionBlock);
                });
                
                formError.classList.add('hidden');
                switchView('questionnaire');
            };

            // Calcula o escore
            const calculateScore = () => {
                const form = document.getElementById('questionnaire-form');
                if (!form.checkValidity()) {
                    formError.classList.remove('hidden');
                    return;
                }
                formError.classList.add('hidden');

                let totalScore = 0;
                patientAnswers = {};
                const formData = new FormData(form);
                
                for (const [key, value] of formData.entries()) {
                    const questionId = parseInt(key.replace('q', ''));
                    const answerValue = parseInt(value);
                    totalScore += answerValue;

                    const question = currentQuestionnaire.conteudo.perguntas.find(p => p.id_pergunta === questionId);
                    const option = currentQuestionnaire.conteudo.opcoes_resposta.find(o => o.valor === answerValue);
                    patientAnswers[questionId] = {
                        text: question.texto_pergunta,
                        answer: option.texto,
                        value: answerValue
                    };
                }

                if (currentQuestionnaire.conteudo.formula_calculo === "SOMA_DAS_RESPOSTAS * 5") {
                    totalScore *= 5;
                }

                renderResults(totalScore);
            };

            // Renderiza os resultados
            const renderResults = (score) => {
                document.getElementById('results-title').textContent = currentQuestionnaire.nome_completo_pt;
                document.getElementById('total-score').textContent = score;

                const interpretation = currentQuestionnaire.conteudo.interpretacao_escore.find(
                    interp => score >= interp.limite_inferior && score <= interp.limite_superior
                );
                
                const interpretationElement = document.getElementById('score-interpretation');
                if (interpretation) {
                    interpretationElement.innerHTML = `<strong class="text-teal-800">${interpretation.classificacao}:</strong> ${interpretation.recomendacao}`;
                } else {
                    interpretationElement.textContent = 'Interpretação não disponível para este escore.';
                }
                
                // Renderizar respostas
                const answersList = document.getElementById('answers-list');
                answersList.innerHTML = '';
                currentQuestionnaire.conteudo.perguntas.forEach((p, index) => {
                    const answer = patientAnswers[p.id_pergunta];
                    if (answer) {
                        answersList.innerHTML += `
                            <div class="border-b pb-1">
                                <p><strong>${index + 1}. ${answer.text}</strong></p>
                                <p class="pl-4"><em>Resposta: ${answer.answer} (Valor: ${answer.value})</em></p>
                            </div>
                        `;
                    }
                });


                switchView('results');
            };

            // Inicialização
            const init = () => {
                renderQuestionnaireList(allQuestionnaires);
                searchInput.addEventListener('input', filterAndSearch);
                filterSelect.addEventListener('change', filterAndSearch);
                backBtn.addEventListener('click', () => switchView('selector'));
                calculateBtn.addEventListener('click', calculateScore);
                newQuestionnaireBtn.addEventListener('click', () => switchView('selector'));
                printBtn.addEventListener('click', () => window.print());
            };

            init();
        });
    </script>
</body>
